CREATE TABLE knihy (
	id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	nazev VARCHAR(255) NOT NULL,
    	autor VARCHAR(155) NOT NULL,
	pocet_stran INT(4) UNSIGNED NOT NULL, 
	pridano TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE ctenari (
	id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    	jmeno VARCHAR(155) NOT NULL,
    	prijmeni VARCHAR(155) NOT NULL,
    	email VARCHAR(155) NOT NULL,
    	registrovan TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE vypujcky (
	id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	ctenar_id INT(11) UNSIGNED NOT NULL,
    	kniha_id INT(11) UNSIGNED NOT NULL,
	vypujceno TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    	navraceno TIMESTAMP 
);
